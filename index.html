function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('BudgetData')
    || ss.insertSheet('BudgetData');
  const val = sheet.getRange('A1').getValue() || '{}';
  const cb  = (e.parameter || {}).callback;
  const body = cb ? cb + '(' + val + ')' : val;
  const mime = cb ? ContentService.MimeType.JAVASCRIPT
                  : ContentService.MimeType.JSON;
  return ContentService.createTextOutput(body).setMimeType(mime);
}
function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('BudgetData')
    || ss.insertSheet('BudgetData');
  sheet.getRange('A1').setValue(e.postData.contents);
  try { updateSummary(ss, JSON.parse(e.postData.contents)); } catch(err) {}
  return ContentService.createTextOutput('{"ok":true}')
    .setMimeType(ContentService.MimeType.JSON);
}
function updateSummary(ss, data) {
  let sum = ss.getSheetByName('Summary')
    || ss.insertSheet('Summary');
  sum.clearContents();
  const hdrs = [['Month','Income','Bills','Subscriptions',
                 'Expenses','Savings Contributed','Ending Balance']];
  sum.getRange(1,1,1,7).setValues(hdrs);
  sum.getRange(1,1,1,7).setFontWeight('bold');
  const months = data.months || {};
  const keys = Object.keys(months).sort().reverse();
  let row = 2;
  keys.forEach(key => {
    const m = months[key]; if (!m) return;
    const ti  = (m.income  ||[]).reduce((s,i)=>s+(+i.amount||0),0);
    const tb  = (m.bills   ||[]).reduce((s,b)=>s+(+b.amount||0),0);
    const ts  = (m.subs    ||[]).reduce((s,x)=>s+(+x.amount||0),0);
    const te  = (m.expenses||[]).reduce((s,x)=>s+(+x.amount||0),0);
    const tsv = (m.savings ||[]).reduce((s,x)=>s+(+x.actual||0),0);
    const end = (+(m.startingBalance||0)) + ti - tb - ts - te - tsv;
    sum.getRange(row,1,1,7).setValues(
      [[key,ti,tb,ts,te,tsv,end]]);
    row++;
  });
  // Format currency columns B-G
  if (row > 2) {
    sum.getRange(2,2,row-2,6)
      .setNumberFormat('$#,##0.00');
  }
}
