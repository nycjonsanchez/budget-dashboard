Connect a Google Sheet so your data syncs across all your devices — free, no extra account needed beyond Google.

Step 1 — Create the Google Sheet

Go to sheets.google.com → create a new blank sheet → name it anything (e.g. "Budget Data").

Step 2 — Add the Apps Script

In your Sheet click Extensions → Apps Script. Delete any existing code, paste the code below, then click Save (💾).

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('BudgetData')
    || ss.insertSheet('BudgetData');
  const val = sheet.getRange('A1').getValue();
  const out = ContentService.createTextOutput(val || '{}');
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('BudgetData')
    || ss.insertSheet('BudgetData');
  sheet.getRange('A1').setValue(e.postData.contents);
  const out = ContentService.createTextOutput('{"ok":true}');
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}
